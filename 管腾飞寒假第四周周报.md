<details>
  <summary><b>硬盘分区</b></summary>

```
 //列出分区情况-p 完整设备名，-f更多信息
lsblk -pf

//接下来要找要用的硬盘，正常应选择独立于Windows外的硬盘
//但我的电脑只有c，d两个盘，只能从d盘中分离
//检查是不是准备的硬盘 -l为详细信息
fdisk -l /dev/。。。

//同样是检查
cfdisk /dev/。。。

//gpt dos sgi sun不同分区表示不同分区模式

//创建100MB的efi system 剩余都变成linux filesystem

//这里没有配置swap是因为有更好的替代
//使用suspend睡眠到内存取代，然后用zram把一部分内存当成交换空间使用

//根目录文件系统分为ext4,btrfs前者稳重，后者有快照，后者更适配arch

//esp需要用fat，因为fat语言更简单

//esp不能挂载到boot中，因为里面有内核等高级工具
//挂载到boot会导致btrfs无法选择地快照boot里的内容

//对ESP文件进行fat32格式化分区，
mkfs.fat -F 32 /dev/...【上述efi文件所在位置】

//对linux fil
mkfs.btrfs /dev/..【上述linux filesystem所在位置】

//跟分区挂载到/mnt中
mount -t btrfs /dev/..[file] /mnt

//创建字卷@和@home
btrfs subvolume create /mnt/@
btrfs subvolume create /mnt/@home

//把跟分区字卷挂载到mnt 「先取消挂载」
//-o表示指定参数 即指定@ 且 透明压缩为zstd，透明压缩主要为节省空间
umount /mnt
mount -t btrfs -o subvol=/@,compress=zstd /dev/..[file] /mnt
mount --mkdir -t btrfs -o subvol=/@home，compress=zstd /dev/..[file] /mnt/home
mount --mkdir /dev/..[efi] /mnt/efi
```
  
</details>

<details>
  <summary><b>安装系统</b></summary>

```
//安装软件到/mnt中，-K表示初始化密钥
//下载软件分别为，基本包，用于编译AUR和软件，性能特调内核，基本的固件btrfs管理工具
pacstrap -K /mnt base base-devel linux-zen linux-firmware btrfs-progs

//网络管理工具，编辑器，管理权限，优化和修复cpu【intel用intel-ucode】
pacstrap /mnt networkmanager vim sudo 【amd-ucode/intel-ucode】

//生成fstab文件,-U 使用UUID指定分区，在fatab文件中写入/mnt挂载
//在启动时会根据这个文件来自动挂载，因为重新启动时上面的挂载会消失
genfstab -U /mnt > /mnt/etc/fstab

//进入新安装的系统
arch-chroot /mnt

```

  
</details>

<details>
  <summary><b>初期配置</b></summary>

```
//创建链接存放在/etc/localtime
ln -s /usr/share/zoneinfo/Asia/Shanghai /etc/localtime

//安装鸭子
pacman -S yazi

//创建系统调整误差的文件
hwclock --systohc

//系统本地化设置
Vim /etc/locale.gen

//删除en_US.UTF-8 UTF和zh_CN.UTF-8 UTF-8前的注释

//生成本地化文件
locale-gen

//编辑文件
vim /etc/locale.conf

//输入LANG=en_US.UTF-8

//编辑主机名
vim /etc/hostname

//输入主机名

//设置密码
passwd

//下载
pacman -S grub efibootmgr

//安装引导 --target指定架构为x86_64 --efi。。。指定ESP位置 --boot。。指定安装目录 --bootloader。。为启动项取名
grub-install --target=x86_64-efi --efi-directory=/efi --boot-directory=/efi --bootloader-id=qi_dong

//由于大部分软件都默认grub在boot上
//所以要创建链接将boot的grub与efi上的grub连接
ln -s /efi/grub /boot/grub

//生成grub的配置文件,并保存到grub.cfg中
grub-mkconfig -o /boot/grub/grub.cfg

//由于原机为windows，所以要配置双系统
//软件功能分别为搜索其他系统，找到windows的efi分区
pacman -S os-prober exfat-utils

//编辑grub源文件
vim /etc/default/grub

//删掉grub_disable_os_prober的注释，允许使用在生成配置文件时使用os-prober搜索其他系统
//删除grub_savedefault的注释，让grub记住启动的是那一项
//把grub_default=后的内容变成saved
//删除日志等级后的quiet并调整等级

//生成grub的配置文件,并保存到grub.cfg中
grub-mkconfig -o /boot/grub/grub.cfg

//配置zram交换空间
pacman -S zram-generator

//编辑配置文件
vim /etc/systemd/zram-generator.conf

//写入
[zram0]
zram-size = ram
compression-algorithm = zstd

//编辑源文件
vim /etc/default/grub

//在内核参数里面加上zswap.enabled=0
//这就是用zram代替zswap

//生成grub的配置文件,并保存到grub.cfg中
grub-mkconfig -o /boot/grub/grub.cfg

//退出chroot环境
exit于原本为Windows系统的电脑来说， 首要步骤应当为磁盘分区

reboot

//之后调整bios就能进系统了
```

//更新

//安装AUR助手
pacman -S yay paru

//分别安装快照snapper 在执行pacman时自动创建快照 图形化交互工具 在grub中添加快照启动项 依赖
